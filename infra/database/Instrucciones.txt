5.- -- 1. Elimina cualquier versión previa (seguro)
	DROP USER IF EXISTS 'handy_user'@'%';

	-- 2. Crea al usuario con plugin compatible
	CREATE USER 'handy_user'@'%' IDENTIFIED WITH mysql_native_password BY 'handy_pass';

	-- 3. Asigna todos los privilegios sobre la base de datos
	GRANT ALL PRIVILEGES ON handy_erp.* TO 'handy_user'@'%' WITH GRANT OPTION;

	-- 4. Refresca los privilegios
	FLUSH PRIVILEGES;

	-- 5. Confirmación (opcional)
	SELECT user, host, plugin FROM mysql.user WHERE user = 'handy_user';
